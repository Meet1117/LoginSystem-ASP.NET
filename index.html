Install NuGet Packages

Install the following essential packages to get started with Entity Framework Core:

Microsoft.EntityFrameworkCore
Microsoft.EntityFrameworkCore.SqlServer
Microsoft.EntityFrameworkCore.Tools

Implementation Tip: Use Package Manager Console with the command: Install-Package PackageName

2

Add Connection String

Configure your database connection in appsettings.json:

{
  "ConnectionStrings": {
    "DefaultConnection": "Server=YOUR_SERVER;Database=LoginSystem_db;Trusted_Connection=True;MultipleActiveResultSets=True;TrustServerCertificate=True"
  }
}

Security Note: Always store production credentials in secure environment variables.

3

Database Dependency Injection

Configure database context dependency injection in Program.cs:

// After builder.Services.AddControllersWithViews();
IConfiguration configuration = new ConfigurationBuilder()
    .SetBasePath(Directory.GetCurrentDirectory())
    .AddJsonFile("appsettings.json")
    .Build();

builder.Services.AddDbContext<DB>(options =>
    options.UseSqlServer(configuration.GetConnectionString("DefaultConnection")));

Place this code after your Authentication setup in Step 4.

4

Add Authentication Dependencies

Configure authentication and session management:

// Authentication Starts
builder.Services.AddAuthentication(CookieAuthenticationDefaults.AuthenticationScheme)
    .AddCookie(option => {
        option.ExpireTimeSpan = TimeSpan.FromMinutes(60);
        option.LoginPath = "/Account/Login";
        option.AccessDeniedPath = "/Account/Login";
    });

builder.Services.AddSession(option => {
    option.IdleTimeout = TimeSpan.FromMinutes(5);
    option.Cookie.HttpOnly = true;
    option.Cookie.IsEssential = true;
});
// Authentication Ends

Place this code after builder.Services.AddControllersWithViews();

Account Controller

Core controller for authentication and user management:

public class AccountController : Controller {
    private readonly DB context;
    public AccountController(DB Context) { this.context = Context; }

    public IActionResult SignUp() => View();

    [HttpPost]
    public IActionResult SignUp(SignUpViewModel model) {
        if (ModelState.IsValid) {
            var data = new User() {
                UserName = model.Username,
                Email = model.Email,
                Password = model.Password,
                Mobile = model.Mobile,
                IsActive = model.IsActive,
            };
            context.Users.Add(data);
            context.SaveChanges();
            TempData["successMessage"] = "Registration successful!";
            return RedirectToAction("Login");
        }
        return View(model);
    }
    
    // ... Login, Logout, and other methods ...
}
